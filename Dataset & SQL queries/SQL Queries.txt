## SQL Queries 

1. Quick preview
SELECT *
FROM employee_attrition
LIMIT 5;

--

2. Attrition values
SELECT attrition, COUNT(*)
FROM employee_attrition
GROUP BY attrition;

--

3.Join Operation
select * 
from employee_attrition as ea
join employee_data as ed
on ea.EmployeeNumber = ed.EmployeeNumber;
